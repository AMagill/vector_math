Code num smoothstep(num edge0, num edge1, num x) #

static num smoothstep(num edge0, num edge1, num x) {
  num t = 0.0;
  t = clamp((x - edge0)/(edge1-edge0), 0.0, 1.0);
  return (t*t)*(3.0-2.0*t);
}

    return -1.0;
